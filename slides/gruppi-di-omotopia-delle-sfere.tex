Vogliamo mostrare che per \(n\ge 3\) dispari e \(i>n\) il gruppo di omotopia \(\pi_i(S^n)\) è finito.
\pause
\begin{itemize}[<+->]
\item[\notdone] \(\pi_i(S^n)\) è finitamente generato.
\item[\notdone] \(\pi_i(S^n)\tensor K=0\) (dove \(K\) denota un campo di caratteristica \(0\)).
\end{itemize}
\end{frame}
\begin{sphereframe}{0}{0}
Sia \(X\) uno spazio topologico semplicemente connesso con gruppi di omologia finitamente generati.
\pause
\begin{block}{Costruzione}
\[
\begin{cases}
X_0=X\\
\visible<+(1)->{T_n=\text{rivestimento universale di \(X_n\)}}\\
\visible<+(1)->{X_{n+1}=\Omega T_n}
\end{cases}
\]
\end{block}
\end{sphereframe}
\begin{sphereframe}{0}{0}
\begin{itemize}
\item \(T_n\to X_n\pause\quad\rightsquigarrow\quad\begin{cases}\pi_i(T_n)=\pi_i(X_n)&i\ge 2\vspace{0.2cm}\pause\\\text{\(E^{p,q}_r\converges H_{p+q}(X_n)\) con}\\ \qquad E^{p,q}_2=H_p(\pi_1(X_n);H_q(T_n))\end{cases}\)
\vspace{0.2cm}
\pause
\item \(\fibration{X_{n+1}}{E}{T_n}\pause\quad\rightsquigarrow\quad\begin{cases}\pi_i(X_{n+1})=\pi_{i+1}(T_n)\vspace{0.2cm}\pause\\\text{\(E^{p,q}_r\converges H_{p+q}(E)\) con}\\ \qquad E^{p,q}_2=H_p(T_n;H_q(X_{n+1}))\end{cases}\)
\end{itemize}
\end{sphereframe}
